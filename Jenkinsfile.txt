pipeline {
    agent any

    stages {
        stage('Code checkout from Git') {
            steps {
                git branch:'master', url:'https://github.com/arlynkatigbak/course-jenkins.git'
            }
        }
        
        stage('Code Quality Check') {
            steps {
                withSonarQubeEnv(installationName : 'my-sonarcloud', credentialsId : 'sonar-token') {
                    sh 'mvn clean verify org.sonarsource.scanner.maven:sonar-maven-plugin:sonar -Dsonar.organization=arlynkatigbak -Dsonar.projectKey=arlynkatigbak_course-jenkins'
                }
            }
        }
        
        stage('Code Security Check') {
            steps {
                sh "chmod a+x mvnw"
                snykSecurity(
                    snykInstallation: 'my-snyk', 
                    snykTokenId:'snyk-token',
                    failOnIssues: false
                )
            }
        }
        
        stage('Build') {
            steps {
                sh 'mvn -B -DskipTests clean package'
            }
        }

        stage('Deploy') {
            steps {
                sh 'set -ex'
                sh 'scp -o StrictHostKeyChecking=no -P 2201 target/jenkins-0.0.1-SNAPSHOT.jar localhost:'
                sh '''
                    ssh -o StrictHostKeyChecking=no localhost -p 2201 '
                    pkill -9 java
                    nohup java -jar -Djava.io.tmpdir=/home/labsuser/.tmp -Dserver.port=8081 jenkins-0.0.1-SNAPSHOT.jar start >> jenkins-lab.log 2>&1 &
                    '
                '''                  
            }
        }

    }
}
